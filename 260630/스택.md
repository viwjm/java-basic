# 📌 문제 이름

스택 만들기

## 🔗 문제 링크

## 💡 문제 해결 전략

큐 만드는 것과 똑같다
단지 인덱스 움직이는 것만 달라질 뿐이다
더 정확히는 push는 똑같고 pop만 다르다

## 💻 코드

```java
public class StackMain {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int arraySize = 5;
        int[] stack = new int[arraySize];
        boolean isEmpty = true;
        int firstIndex = 0;
        int lastIndex = 0;

        System.out.println("명령 수를 입력하세요:");
        int numOrders = scanner.nextInt();
        scanner.nextLine();

        for (int i = 0; i < numOrders; i++) {
            String order = scanner.nextLine();

            if (isPush(order)) {
                if (isFull(firstIndex, lastIndex, arraySize)) {
                    System.out.println("배열 꽉 참");

                    continue;
                }

                int number = Integer.parseInt(order.split(" ")[1]);

                if (isEmpty) {
                    stack[lastIndex] = number;
                    isEmpty = false;
                } else {
                    lastIndex = (lastIndex + 1) % arraySize;
                    stack[lastIndex] = number;
                }

                continue;
            }

            if (order.equals("pop")) {
                if (isEmpty) {
                    System.out.println("-1");

                    continue;
                }

                System.out.println(stack[lastIndex]);
                stack[lastIndex] = 0;

                if (firstIndex == lastIndex) {
                    isEmpty = true;
                } else {
                    lastIndex = (lastIndex + 5 - 1) % arraySize;
                }

                continue;
            }

            if (order.equals("size")) {
                if (isEmpty) {
                    System.out.println("0");

                    continue;
                }

                int size = firstIndex <= lastIndex ? lastIndex - firstIndex + 1 : lastIndex + 5 - firstIndex + 1;
                System.out.println(size);

                continue;
            }

            if (order.equals("empty")) {
                if (isEmpty) {
                    System.out.println("1");

                    continue;
                }

                System.out.println("0");

                continue;
            }

            if (order.equals("front")) {
                if (isEmpty) {
                    System.out.println("-1");

                    continue;
                }

                System.out.println(stack[firstIndex]);

                continue;
            }

            if (order.equals("back")) {
                if (isEmpty) {
                    System.out.println("-1");

                    continue;
                }

                System.out.println(stack[lastIndex]);

                continue;
            }

            System.out.println("잘못된 입력입니다");
        }
    }

    private static boolean isPush(String order) {
        String[] orderComponents = order.split(" ");

        if (orderComponents.length != 2) {
            return false;
        }

        try {
            Integer.parseInt(orderComponents[1]);
        } catch (NumberFormatException e) {
            return false;
        }

        return orderComponents[0].equals("push");
    }

    private static boolean isFull(int firstIndex, int lastIndex, int arraySize) {
        return firstIndex == (lastIndex + 1) % arraySize;
    }
}
```
